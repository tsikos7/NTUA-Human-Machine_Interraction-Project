1. Where are we?
uname -a

2. Become superuser
sudo su
# and enter password
# HINT: type exit to exit superuser shell

# HINT:
# copy from the terminal: Windows & Putty: select text / Unix: select text & hit Ctrl+Shift+C
# paste to the terminal  Windows & Putty: δεξί κλικ / Unix: Ctrl+Shift+V

3. optional: upgrade all packages
apt-get update
apt-get upgrade

4. Install Apache
apt-get install apache2 apache2-utils
# hit enter (yes) to continue

5. Start and enable (at system boot time) Apache2 web server
systemctl start apache2
systemctl enable apache2
# Check that we have a web server: open a browser and go to your hostname!

6. In a webrowser go to https://www.random.org/passwords/ and create 5 random passwords 10 characters long. Leave this tab open

7. Install MySQL Database
apt-get install mysql-client mysql-server
# you will be prompted for the root mysql password. enter the 1st random password and save it to your text file

8. Install PHP, PHP modules and Apache PHP modules
apt-get install php7.0 php7.0-mysql libapache2-mod-php7.0 php7.0-cli php7.0-cgi php7.0-gd

9. Download latest Wordpress and unpack
wget -c http://wordpress.org/latest.tar.gz
tar -xzvf latest.tar.gz

10. Arrange file locations and permissions
cd /var/www/html/
mv /home/user/wordpress .
mv wordpress/ site1
sudo chown -R www-data:www-data site1/

11. Create a WP database for site1
mysql -u root -p
#enter root MySQL database password
#you are now inside mysql client
CREATE DATABASE wp_site1;
GRANT ALL PRIVILEGES ON wp_site1.* TO 'wp_site1_dbuser'@'localhost' IDENTIFIED BY '%%%%%%%%%%';
#IDENTIFIED BY '%%%%%%%%%%' is the next random password generated by RANDOM.ORG. Save it to your text file
FLUSH PRIVILEGES;
EXIT;
#quit mysql client

12. Configure WP Database connection
cd site1
#we are now in /var/www/html/site1
mv wp-config-sample.php wp-config.php
vim wp-config.php

Make the following (3) changes
// ** MySQL settings - You can get this info from your web host ** //
/** The name of the database for WordPress */
define('DB_NAME', 'wp_site1');
/** MySQL database username */
define('DB_USER', 'wp_site1_dbuser');
/** MySQL database password */
define('DB_PASSWORD', '%%%%%%%%%%');


13. Restart services
systemctl restart apache2
systemctl restart mysql

14. Goto http://hostname/site1
